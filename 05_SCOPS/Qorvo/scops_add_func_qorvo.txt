FUNC_CHECK_RCD

global widget

global g_hostname
global g_lotname
global g_dcc
global g_operation
global g_operatorid
global g_customer
global g_job

set qry "SCOPS,CALLSP2,CheckQorvoRCD,CHECK,$g_hostname,$g_lotname,$g_dcc,$g_operation,$g_operatorid,$g_customer,$g_job,,0*"

set res ""
set res [FUNC_SEND_QRY $qry]

if { [lindex [split $res ","] 0] == "FAIL" || $res == "" } {
    set choice [ tk_messageBox -title "QORVO RCD CHECK ERROR" -icon warning -message [FM "TESTIT" "" $res] ]
    return "FALSE"
}

if { [lindex [split $res ","] 0] == "NO" } {
    return "TRUE"
}

set rcd_path [lindex [split $res ","] 0]

set total_count ""
set total_count [lindex [split $res ","] 2]
set now_count ""
set extension ""
set extension [lindex [split $res ","] 1]
set file_exist "Yes"

### RCD Path Check ###
for {set i 0} {$i < $total_count} {incr i} {
    set now_count $i
    set file_exist "Yes"
    
    if { [file exist $rcd_path] == 1 } {
        if {$i == $total_count - 1} {
            if [catch {
                set gt1_list [glob -directory $rcd_path $extension]
            } err] {
                set win_title_file "qorvo_rcd_file_check_warning"
                FUNC_DISPLAY_MESSAGE "close" "$win_title_file" "" ""
                set config_file "$win_title_file"           
    
                FUNC_DISPLAY_MESSAGE "open" $config_file "image,scops_warning_program.gif" [FM "K3 Engineer" "Qorvo RCD Path, There is no RCD File!" ""]
                set choice [ tk_messageBox -title "QORVO RCD FILE CHECK ERROR" -icon error -message [FM "K3 Engineer" "Qorvo RCD File, There is no RCD File!" ""] ]
                FUNC_DISPLAY_MESSAGE "close" "$win_title_file" "" ""
                return "FALSE"  
            }  
            set file_exist "Yes"  
        }
        
        if [catch {
                set gt1_list [glob -directory $rcd_path $extension]
            } err] {
                set file_exist "No"
            }   
        
        if {$file_exist == "Yes"} {
            break
        }                  
    }
    
    if {$i == $total_count - 1} {
        set win_title_path "qorvo_rcd_path_check_warning"
        FUNC_DISPLAY_MESSAGE "close" "$win_title_path" "" ""
        set config_path "$win_title_path"
    
        FUNC_DISPLAY_MESSAGE "open" $config_path "image,scops_warning_program.gif" [FM "K3 Engineer" "Qorvo RCD Path, No exist!" ""]
        set choice [ tk_messageBox -title "QORVO RCD PATH CHECK ERROR" -icon error -message [FM "K3 Engineer" "Qorvo RCD Path, No exist!" $res] ]
        FUNC_DISPLAY_MESSAGE "close" "$win_title_path" "" ""
        return "FALSE"  
    }
    
    set now_count [expr $i+1]
    set qry "SCOPS,CALLSP2,CheckQorvoRCD,CHECK,$g_hostname,$g_lotname,$g_dcc,$g_operation,$g_operatorid,$g_customer,$g_job,,$now_count*"
    set res [FUNC_SEND_QRY $qry]
    set rcd_path [lindex [split $res ","] 0]
    set extension [lindex [split $res ","] 1]
    
}

set gt1_list_count [llength $gt1_list]
    
if { $gt1_list_count != 1 } {
    set win_title_files "qorvo_rcd_file_check_warning"
    FUNC_DISPLAY_MESSAGE "close" "$win_title_files" "" ""
    set config_files "$win_title_files"
    
    FUNC_DISPLAY_MESSAGE "open" $config_files "image,scops_warning_program.gif" [FM "K3 Engineer" "Qorvo RCD Path, There are many RCD Files!" ""]
    set choice [ tk_messageBox -title "QORVO RCD FILE CHECK ERROR" -icon error -message [FM "K3 Engineer" "Qorvo RCD File, There are many RCD Files!\n\nTotal Count:" $gt1_list_count] ]
    FUNC_DISPLAY_MESSAGE "close" "$win_title_files" "" ""
    return "FALSE"  
        
}

### validation ###

set qry_val "SCOPS,CALLSP2,CheckQorvoRCD,VALIDATION,$g_hostname,$g_lotname,$g_dcc,$g_operation,$g_operatorid,$g_customer,$g_job,$gt1_list,$now_count*"

set res_val ""
set res_val [FUNC_SEND_QRY $qry_val]
    
if { [lindex [split $res_val ","] 0] == "FAIL" || $res_val == "" } {
    set choice [ tk_messageBox -title "QORVO RCD CHECK ERROR" -icon warning -message [FM "TESTIT" "" $res_val] ]
    return "FALSE"
}

if { [lindex [split $res_val ","] 0] == "YES" } {
    
    set msg [lindex [split $res_val ","] 1]
    
    set win_title_val "qorvo_rcd_file_validation_warning"
    FUNC_DISPLAY_MESSAGE "close" "$win_title_val" "" ""
    set config_val "$win_title_val"
    
    FUNC_DISPLAY_MESSAGE "open" $config_val "image,scops_warning_program.gif" [FM "K3 Engineer" "Qorvo RCD Validation False!" ""]
    set choice [ tk_messageBox -title "QORVO RCD FILE VALIDATION ERROR" -icon error -message [FM "K3 Engineer" $msg ""] ]
    FUNC_DISPLAY_MESSAGE "close" "$win_title_val" "" ""
    return "FALSE"  
    
}

return "TRUE"
